from odoo import api, fields, models

class HospitalPatient(models.Model):
    _name = 'hospital.patient'
    _description = 'this is make for store hospital patient data'
    _inherit = ['mail.thread', 'mail.activity.mixin']

    name = fields.Char(string="Name", required=True)
    dob = fields.Datetime(string="Data Of Birth")
    gender = fields.Selection([('male', 'Male'), ('female', 'Female')], string="Gender")
    stage_id = fields.Many2one(
        'hospital.patient.stage',
        string="Stage",
        group_expand='_read_group_stage_ids',  # ← এটা লাগবে
    )

    @api.model
    def _read_group_stage_ids(self, stages, domain):
        return stages.search([])


class HospitalPatientStage(models.Model):
    _name = 'hospital.patient.stage'
    _fold_name = 'fold'              # ← এটা লাগবে
    _order = 'sequence, id'

    name = fields.Char(string="Stage Name", required=True)
    sequence = fields.Integer(default=10)
    fold = fields.Boolean(string="Folded")
