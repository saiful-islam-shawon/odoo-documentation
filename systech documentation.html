one2may :
Step 1: Child Model (must have Many2one back-reference)

# models/book_list.py
from odoo import models, fields

class BookList(models.Model):
    _name = 'book.list'
    _description = 'Book List'

    # ✅ Many2one pointing back to parent
    bookmark_id = fields.Many2one(
        'zc.bookmark.history',
        string="Bookmark",
        ondelete='cascade'  # delete lines when parent deleted
    )

    product_id = fields.Many2one('product.product', string="Product")
    amount = fields.Float(string="Amount")
    qty = fields.Float(string="Quantity")
    per_unit_id = fields.Many2one('uom.uom', string="Per Unit")

Step 2: Parent Model (One2many field)

# models/zc_bookmark_history.py
from odoo import models, fields

class ZcBookmarkHistory(models.Model):
   _name = 'zc.bookmark.history'
   _description = 'Bookmark History'

   name = fields.Char(string="Name")
   status = fields.Selection([
       ('draft', 'Draft'),
       ('validated', 'Validated'),
   ], default='draft')

   # ✅ One2many field
   book_line_ids = fields.One2many(
       'book.list',        # child model name
       'bookmark_id',      # Many2one field in child model
       string="Books"
   )

Step 3: XML View

<form>
    <header>
        <button name="action_validated"
                string="Validate"
                type="object"
                invisible="status == 'validated'"/>
        <field name="status" widget="statusbar"/>
    </header>
    <sheet>
        <group>
            <field name="name"/>
        </group>

        <notebook>
            <page name="book_page" string="Books">
                <!-- ✅ One2many shown as editable list -->
                <field name="book_line_ids">
                    <list editable="bottom">
                        <field name="product_id"
                               readonly="parent.status == 'validated'"/>
                        <field name="amount"/>
                        <field name="qty"/>
                        <field name="per_unit_id"/>
                    </list>
                </field>
            </page>
        </notebook>
    </sheet>
</form>
