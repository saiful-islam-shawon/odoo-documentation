static dashboard creation

প্রথমে OWL এবং Dashboard দেখানোর জন্য একটি Menu তৈরি করতে হবে।
views/menu.xml

<?xml  version="1.0"  encoding="utf-8"?>

<odoo>

<menuitem
id="dashboard_menu_root"
name="DMR"
sequence="1"/>

<menuitem
id="dashboard_menu"
name="Dashboard"
sequence="1"
parent="dashboard_menu_root"
action="sales_dashboard_action"
/>

</odoo>

এখন একটি Client Action তৈরি করতে হবে।

<?xml  version="1.0"  encoding="utf-8"?>
<odoo>
<record  id="sales_dashboard_action"  model="ir.actions.client">
	<field  name="name">Sales Dashboard</field>
	<field  name="tag">owl.sales_dashboard</field>
</record>
</odoo>

এখন static/src/components ফোল্ডারের ভিতরে component এর জন্য একটি XML এবং JS ফাইল তৈরি করতে হবে।
static/src/components/sales_dashboard.js

/** @odoo-module */

import { registry } from "@web/core/registry";
import { Component, onWillStart, useRef, onMounted } from "@odoo/owl";
import { KpiCard } from "./kpi_card/kpi_card";
import { loadJS } from "@web/core/assets";

export class OwlSalesDashboard extends Component {
    setup() {
        this.chartref = useRef("chart")
        onWillStart(async () => {
            await loadJS("https://cdnjs.cloudflare.com/ajax/libs/Chart.js/4.5.0/chart.umd.min.js")
        })

        onMounted(() => {
            const data = [
                { year: 2010, count: 10 },
                { year: 2011, count: 20 },
                { year: 2012, count: 15 },
                { year: 2013, count: 25 },
                { year: 2014, count: 22 },
                { year: 2015, count: 30 },
                { year: 2016, count: 28 },
            ];

            new Chart(
                this.chartref.el,
                {
                    type: 'pie',
                    data: {
                        labels: [
                            'Red',
                            'Blue',
                            'Yellow'
                        ],
                        datasets: [{
                            label: 'My First Dataset',
                            data: [300, 50, 100],
                            backgroundColor: [
                                'rgb(255, 99, 132)',
                                'rgb(54, 162, 235)',
                                'rgb(255, 205, 86)'
                            ],
                            hoverOffset: 4
                        }]
                    }
                }
            );
        })
    }
}

OwlSalesDashboard.template = "owl.OwlSalesDashboard";
OwlSalesDashboard.components = { KpiCard }

registry.category("actions").add("owl.sales_dashboard", OwlSalesDashboard);
