step 1: controler.py

from odoo import http
from odoo.http import request


class ProductFormController(http.Controller):

  @http.route('/product/form',type='http',auth='public',website=True,methods=['GET']
  )
  def product_form(self, success=None, **kwargs):

      products = request.env['product.product'].sudo().search([])

      values = {
          'products': products,
          'success': success,
      }

      return request.render('practice_module.product_form_template',values)

  @http.route('/product/submit',type='http',auth='public',website=True,methods=['POST'],csrf=True)
  def submit_inquiry(self, **post):

      product_id = int(post.get('product_id', 0))

      request.env['product.inquiry'].sudo().create({
          'name': post.get('name'),
          'email': post.get('email'),
          'product_id': product_id,
          'message': post.get('message'),
      })

      return request.redirect('/product/form?success=1')
