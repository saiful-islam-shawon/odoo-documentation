model

from odoo import models, fields

class PracticeModel(models.Model):
    _name = "practice.model"
    _description = "Practice Model"

    name = fields.Char("Name")
    company_id = fields.Many2one(
        "res.company",
        string="Company",
        default=lambda self: self.env.company,
        index=True
    )

Record Rule

<record id="rule_practice_model_multi_company" model="ir.rule">
