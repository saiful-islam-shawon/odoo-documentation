step 1: create controlers folder
controlers/init.py
controlers/controler.py

from odoo import http
from odoo.http import request

class ProductCatalogControler(http.Controller):
  @http.route('/product/catalog/', auth='public', website=True)
  def product_catalog(self, **kwargs):
    products = request.env['product.template'].search([])
    return request.render('practice_module.product_catalog_template', {'products': products})
