How to Create Stat Button

Step 1 — Add in Form View

<form>
    <!-- Stat Buttons -->
    <div class="oe_button_box" name="button_box">

        <button type="object"
            name="action_get_appointments"
            class="oe_stat_button"
            icon="fa-calendar">
            <field name="appointment_count" string="Appointments" widget="statinfo"/>
        </button>

    </div>

    <sheet>
        <group>
            <field name="name"/>
        </group>
    </sheet>
</form>

Step 2 — Add Count Field in Model

from odoo import api, fields, models

class Patient(models.Model):
    _name = 'hospital.patient'

    name = fields.Char(string='Name')

    # count field
    appointment_count = fields.Integer(
        string='Appointments',
        compute='_compute_appointment_count',
    )

    # compute count
    @api.depends('appointment_ids')
    def _compute_appointment_count(self):
        for rec in self:
            rec.appointment_count = self.env['hospital.appointment'].search_count([
                ('patient_id', '=', rec.id)
            ])

    # button click action
    def action_get_appointments(self):
        return {
            'type': 'ir.actions.act_window',
            'name': 'Appointments',
            'res_model': 'hospital.appointment',
            'view_mode': 'list,form',
            'domain': [('patient_id', '=', self.id)],
        }
